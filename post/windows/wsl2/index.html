WSL2 相关配置

.wslconfig

[wsl2]
# 处理器核心数
processors=4
# 最大运行内存
memory=4GB
# 交换分区大小
swap=8GB

[experimental]
# WSL2 虚拟机内存自动回收策略，主动回收页缓存等系统缓存，回收力度强于gradual（默认值）
autoMemoryReclaim=dropCache

/etc/wsl.conf

[boot]
# 开启WSL2的Systemd服务管理
systemd=true

[interop]
# 禁止将Windows宿主机的PATH路径追加到WSL2中
appendWindowsPath=false

手动安装Ubuntu设置默认用户

echo -e "[user]\ndefault=banyejiu" > /etc/wsl.conf
