# 一、Docker安装密码管理器Vaultwarden

# 1.1 创建专用网络

docker network create middleware

# 1.2 准备工作

mkdir -p /mydata/vaultwarden/data
mkdir -p /mydata/nginx/conf
mkdir -p /mydata/nginx/ssl

# 1.3 安装vaultwarden

docker run -d \
  --name vaultwarden \
  -e INVITATIONS_ALLOWED=false \
  -e SHOW_PASSWORD_HINT=false \
  -e LOG_FILE=/data/bitwarden.log \
  -e TZ=Asia/Shanghai \
  -p 8001:80 \
  --network middleware \
  -v /mydata/vaultwarden/data:/data/ \
  --log-driver json-file \
  --log-opt max-size=200k \
  --log-opt max-file=10 \
  vaultwarden/server:latest

# 1.3.1 环境变量设置

禁用密码提示添加环境变量 SHOW_PASSWORD_HINT=false 禁用邀请添加环境变量 INVITATIONS_ALLOWED=false 启用日志记录添加环境变量 LOG_FILE=/data/bitwarden.log 禁用新用户注册添加环境变量 SIGNUPS_ALLOWED=false 禁用web管理页面添加环境变量 WEB_VAULT_ENABLED=false 开启随机路径添加环境变量 DOMAIN=https://二级域名:端口(cloudflare支持的端口)/随机路径 日志级别默认为 “info”，可以参考 bitwarden_rs wiki 更改日志级别。

如果想要开启管理页面添加环境变量:ADMIN_TOKEN=token str
